## ğŸ¨ Visual Pattern Showcase

### 1. Solid Star Square

```cpp
for (int i = 1; i <= n; i++) {                //  Output (for n=4):
    for (int j = 1; j <= n; j++) {            //  * * * * 
        cout<<"* ";                           //  * * * * 
    }                                        //   * * * * 
    cout<<"\n";                              //   * * * * 
}
```

### 2. Number Triangle

```cpp
for (int i = 1; i <= n; i++) {                //  Output (for n=4):
    for (int j = 1; j <= i; j++) {            //  1 
        cout<<j;                              //  1 2 
    }                                        //   1 2 3 
    cout<<"\n";                              //   1 2 3 4 
}
```

### 3. Alphabet Pyramid

```cpp
for (int i = 1; i <= n; i++) {                //  Output (for n=4):
    char ch = 'A';                            //  A 
    for (int j = 1; j <= i; j++) {            //  A B 
        cout<<ch++;                           //  A B C 
    }                                        //   A B C D 
    cout<<"\n";                              //   A B C D E 
}
```

### 4. Hollow Rectangle

```cpp
for (int i = 1; i <= rows; i++) {                             
    for (int j = 1; j <= cols; j++) {
        if (i == 1 || i == rows || j == 1 || j == cols)
            cout<<"* ";
        else
            cout<<"  ";
    }
    cout<<"\n";
}
```

### 5. Mirrored Left Triangle

```cpp
for (int i = 1; i <= n; i++) {
    for (int j = 1; j <= n - i; j++) cout<<"  ";
    for (int k = 1; k <= i; k++) cout<<"* ";
    cout<<"\n";
}
```
